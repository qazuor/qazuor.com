```typescript
export const isPlainObject = (value: unknown): value is Record<string, unknown> =>
  Object.prototype.toString.call(value) === '[object Object]';

// Usage
isPlainObject({});        // true
isPlainObject({ a: 1 });  // true
isPlainObject([]);        // false
isPlainObject(null);      // false
isPlainObject(new Date);  // false
```
