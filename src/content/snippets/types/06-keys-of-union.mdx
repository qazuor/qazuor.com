```typescript
export type KeysOfUnion<T> = T extends T ? keyof T : never;

// Usage
type A = { a: string; shared: number };
type B = { b: boolean; shared: number };

type AllKeys = KeysOfUnion<A | B>;
// 'a' | 'b' | 'shared'
```
