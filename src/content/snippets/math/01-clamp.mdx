```typescript
export const clamp = (value: number, min: number, max: number) =>
  Math.min(max, Math.max(min, value));

// Usage
clamp(5, 0, 10);   // 5
clamp(-5, 0, 10);  // 0
clamp(15, 0, 10);  // 10
```
