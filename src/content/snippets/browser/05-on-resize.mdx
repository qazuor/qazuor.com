```typescript
export const onResize = (fn: () => void) => {
  window.addEventListener('resize', fn);
  fn();
  return () => window.removeEventListener('resize', fn);
};

// Usage
const cleanup = onResize(() => {
  console.log('Width:', window.innerWidth);
});

// Later, to unsubscribe:
cleanup();
```
